        <line x1="6" y1="6" x2="18" y2="18"></line>

      </svg>

    );


    const Plus = ({ className, size = 24 }) => (

      <svg className={className} width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">

    // Main ChoreTracker Component with Firebase integration

    const ChoreTracker = () => {

      const [chores, setChores] = useState([]);

      const [changeLog, setChangeLog] = useState([]);

      const [activeTab, setActiveTab] = useState('today');

      const [showAddModal, setShowAddModal] = useState(false);

      const [editingChore, setEditingChore] = useState(null);

      const [editingUpcomingDate, setEditingUpcomingDate] = useState('');

      const [expandedCalendarChoreId, setExpandedCalendarChoreId] = useState(null);

      const [loading, setLoading] = useState(true);


      // Form state

      const [newChore, setNewChore] = useState({

        name: '',

        frequency: 'daily',

        dueDate: '',

        recurrence: 'everyday'

      });


      // Load data from Firebase on mount

      useEffect(() => {

        const loadData = async () => {

          try {

            // Load chores

            const choresSnapshot = await db.collection('chores').get();

            const loadedChores = choresSnapshot.docs.map(doc => ({

              id: doc.id,

              ...doc.data()

            }));

            setChores(loadedChores);


            // Load change log

            const logSnapshot = await db.collection('changeLog')

              .orderBy('timestamp', 'desc')

              .limit(50)

              .get();

            const loadedLog = logSnapshot.docs.map(doc => ({

              id: doc.id,

              ...doc.data()

            }));

            setChangeLog(loadedLog);

          } catch (error) {

            console.error('Error loading data:', error);

          } finally {

            setLoading(false);

          }

        };


        loadData();

      }, []);


      // Helper function to save chores to Firebase

      const saveChore = async (chore) => {

        try {

          if (chore.id) {

            // Update existing chore

            await db.collection('chores').doc(chore.id).set(chore);

          } else {

            // Add new chore

            const docRef = await db.collection('chores').add(chore);

            return { ...chore, id: docRef.id };

          }

        } catch (error) {

          console.error('Error saving chore:', error);

        }

      };


      // Helper function to save change log entry

      const saveLogEntry = async (entry) => {

        try {

          await db.collection('changeLog').add(entry);

        } catch (error) {

          console.error('Error saving log entry:', error);

        }

      };


      // Helper function to delete chore from Firebase

      const deleteChoreFromDB = async (choreId) => {

        try {

          await db.collection('chores').doc(choreId).delete();

        } catch (error) {

          console.error('Error deleting chore:', error);

        }

      };


      // Get Hawaii date

      const getHawaiiDate = () => {

        const hawaiiTime = new Date().toLocaleString("en-US", {

          timeZone: "Pacific/Honolulu"

        });

        return new Date(hawaiiTime);

      };


      const formatDate = (date) => {

        const d = new Date(date);

        const month = String(d.getMonth() + 1).padStart(2, '0');

        const day = String(d.getDate()).padStart(2, '0');

        const year = d.getFullYear();

        return `${month}/${day}/${year}`;

      };


      const isToday = (date) => {

        const today = getHawaiiDate();

        const checkDate = new Date(date);

        return today.toDateString() === checkDate.toDateString();

      };


      const isPast = (date) => {

        const today = getHawaiiDate();

        today.setHours(0, 0, 0, 0);

        const checkDate = new Date(date);

        checkDate.setHours(0, 0, 0, 0);

        return checkDate < today;

      };


      const addChore = async () => {

        if (newChore.name && newChore.dueDate) {

          const choreToAdd = {

            name: newChore.name,

            frequency: newChore.frequency,

            dueDate: newChore.dueDate,

            recurrence: newChore.recurrence,

            completed: false,

            completedBy: null,

            completedDate: null

          };


          const savedChore = await saveChore(choreToAdd);

          

          if (savedChore) {

            setChores([...chores, savedChore]);

            

            const logEntry = {

              action: 'added',

              choreName: newChore.name,

              person: 'System',

              timestamp: getHawaiiDate().toISOString(),

              details: `Added "${newChore.name}" due on ${formatDate(newChore.dueDate)}`

            };

            

            await saveLogEntry(logEntry);

            setChangeLog([logEntry, ...changeLog]);

          }


          setNewChore({ name: '', frequency: 'daily', dueDate: '', recurrence: 'everyday' });

          setShowAddModal(false);

        }

      };


      const markAsComplete = async (choreId, person) => {

        const updatedChores = chores.map(chore => {

          if (chore.id === choreId) {

            let nextDueDate = null;

            const currentDue = new Date(chore.dueDate);


            if (chore.frequency === 'daily') {

              if (chore.recurrence === 'everyday') {

                nextDueDate = new Date(currentDue);

                nextDueDate.setDate(nextDueDate.getDate() + 1);

              } else if (chore.recurrence === 'weekdays') {

                nextDueDate = new Date(currentDue);

                do {

                  nextDueDate.setDate(nextDueDate.getDate() + 1);

                } while (nextDueDate.getDay() === 0 || nextDueDate.getDay() === 6);

              }

            } else if (chore.frequency === 'weekly') {

              nextDueDate = new Date(currentDue);

              nextDueDate.setDate(nextDueDate.getDate() + 7);

            }


            const updatedChore = {

              ...chore,

              completed: true,

              completedBy: person,

              completedDate: getHawaiiDate().toISOString(),

              dueDate: nextDueDate ? nextDueDate.toISOString().split('T')[0] : chore.dueDate

            };


            saveChore(updatedChore);


            const logEntry = {

              action: 'completed',

              choreName: chore.name,

              person: person,

              timestamp: getHawaiiDate().toISOString(),

              details: nextDueDate 

                ? `Next due date: ${formatDate(nextDueDate)}` 

                : 'No recurrence set'

            };

            

            saveLogEntry(logEntry);

            setChangeLog([logEntry, ...changeLog]);


            return updatedChore;

          }

          return chore;

        });


        setChores(updatedChores);

      };


      const deleteChore = async (choreId) => {

        const choreToDelete = chores.find(c => c.id === choreId);

        

        await deleteChoreFromDB(choreId);

        setChores(chores.filter(c => c.id !== choreId));

        

        const logEntry = {

          action: 'deleted',

          choreName: choreToDelete.name,

          person: 'System',

          timestamp: getHawaiiDate().toISOString(),

          details: `Deleted "${choreToDelete.name}"`

        };

        

        await saveLogEntry(logEntry);

        setChangeLog([logEntry, ...changeLog]);

      };


      const updateDueDate = async (choreId, newDate) => {

        const updatedChores = chores.map(chore => {

          if (chore.id === choreId) {

            const updatedChore = {

              ...chore,

              dueDate: newDate,

              completed: false,

              completedBy: null,

              completedDate: null

            };

            

            saveChore(updatedChore);

            

            const logEntry = {

              action: 'edited',

              choreName: chore.name,

              person: 'System',

              timestamp: getHawaiiDate().toISOString(),

              details: `Rescheduled "${chore.name}" to be due on ${formatDate(newDate)}`

            };

            

            saveLogEntry(logEntry);

            setChangeLog([logEntry, ...changeLog]);

            

            return updatedChore;

          }

          return chore;

        });


        setChores(updatedChores);

        setEditingUpcomingDate('');

        setExpandedCalendarChoreId(null);

      };


      // Statistics

      const stats = useMemo(() => {

        const today = getHawaiiDate();

        const todayStr = today.toISOString().split('T')[0];

        

        const todayChores = chores.filter(c => c.dueDate === todayStr);

        const completed = todayChores.filter(c => c.completed).length;

        const total = todayChores.length;

        

        const kawikaTasks = todayChores.filter(c => c.completedBy === 'Kawika').length;

        const jordynTasks = todayChores.filter(c => c.completedBy === 'Jordyn').length;


        return {

          completed,

          total,

          percentage: total > 0 ? Math.round((completed / total) * 100) : 0,

          kawika: kawikaTasks,

          jordyn: jordynTasks

        };

      }, [chores]);


      // Filter chores by tab

      const filteredChores = useMemo(() => {

        const today = getHawaiiDate();

        const todayStr = today.toISOString().split('T')[0];


        if (activeTab === 'today') {

          return chores.filter(c => c.dueDate === todayStr);

        } else if (activeTab === 'upcoming') {

          return chores.filter(c => !isToday(c.dueDate) && !isPast(c.dueDate));

        } else if (activeTab === 'history') {

          return chores.filter(c => c.completed || isPast(c.dueDate));

        }

        return chores;

      }, [chores, activeTab]);


      if (loading) {

        return (

          <div className="min-h-screen bg-gradient-to-br from-blue-50 to-purple-50 flex items-center justify-center">

            <div className="text-xl font-semibold text-gray-700">Loading...</div>

          </div>

        );

      }


      return (

        <div className="min-h-screen bg-gradient-to-br from-blue-50 to-purple-50 p-4">

          <div className="max-w-4xl mx-auto">

            {/* Header */}

            <div className="bg-white rounded-3xl shadow-lg p-6 mb-6">

              <h1 className="text-4xl font-bold text-gray-800 mb-2">Lopez Family Chores</h1>

              <p className="text-gray-600">Track and manage household tasks</p>

            </div>


            {/* Stats Cards */}

            <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-6">

              <div className="bg-white rounded-2xl shadow-md p-6">

                <div className="flex items-center justify-between mb-2">

                  <span className="text-gray-600 font-medium">Today's Progress</span>

                  <CheckCircle className="text-teal-500" size={24} />

                </div>

                <div className="text-3xl font-bold text-gray-800">{stats.percentage}%</div>

                <div className="text-sm text-gray-500 mt-1">{stats.completed} of {stats.total} completed</div>

              </div>


              <div className="bg-white rounded-2xl shadow-md p-6">

                <div className="flex items-center justify-between mb-2">

                  <span className="text-gray-600 font-medium">Kawika</span>

                  <TrendingUp className="text-blue-500" size={24} />

                </div>

                <div className="text-3xl font-bold text-gray-800">{stats.kawika}</div>

                <div className="text-sm text-gray-500 mt-1">tasks completed today</div>

              </div>


              <div className="bg-white rounded-2xl shadow-md p-6">

                <div className="flex items-center justify-between mb-2">

                  <span className="text-gray-600 font-medium">Jordyn</span>

                  <TrendingUp className="text-purple-500" size={24} />

                </div>

                <div className="text-3xl font-bold text-gray-800">{stats.jordyn}</div>

                <div className="text-sm text-gray-500 mt-1">tasks completed today</div>

              </div>

            </div>


            {/* Tabs */}

            <div className="bg-white rounded-2xl shadow-md p-2 mb-6 flex gap-2">

              <button

                onClick={() => setActiveTab('today')}

                className={`flex-1 py-3 px-4 rounded-xl font-medium transition ${

                  activeTab === 'today' 

                    ? 'bg-teal-500 text-white' 

                    : 'text-gray-600 hover:bg-gray-100'

                }`}

              >

                Today

              </button>

              <button

                onClick={() => setActiveTab('upcoming')}

                className={`flex-1 py-3 px-4 rounded-xl font-medium transition ${

                  activeTab === 'upcoming' 

                    ? 'bg-teal-500 text-white' 

                    : 'text-gray-600 hover:bg-gray-100'

                }`}

              >

                Upcoming

              </button>

              <button

                onClick={() => setActiveTab('history')}

                className={`flex-1 py-3 px-4 rounded-xl font-medium transition ${

                  activeTab === 'history' 

                    ? 'bg-teal-500 text-white' 

                    : 'text-gray-600 hover:bg-gray-100'

                }`}

              >

                History

              </button>

            </div>


            {/* Chores List */}

            <div className="bg-white rounded-2xl shadow-md p-6 mb-6">

              <div className="flex justify-between items-center mb-4">

                <h2 className="text-2xl font-bold text-gray-800">

                  {activeTab === 'today' && 'Today\'s Tasks'}

                  {activeTab === 'upcoming' && 'Upcoming Tasks'}

                  {activeTab === 'history' && 'History'}

                </h2>

                <button

                  onClick={() => setShowAddModal(true)}

                  className="bg-teal-500 text-white py-2 px-4 rounded-xl font-medium hover:bg-teal-600 transition flex items-center gap-2"

                >

                  <Plus size={20} />

                  Add Chore

                </button>

              </div>


              {filteredChores.length === 0 ? (

                <div className="text-center py-12 text-gray-500">

                  No tasks {activeTab === 'today' ? 'for today' : activeTab === 'upcoming' ? 'upcoming' : 'in history'}

                </div>

              ) : (

                <div className="space-y-3">

                  {filteredChores.map(chore => (

                    <div

                      key={chore.id}

                      className={`p-4 rounded-xl border-2 ${

                        chore.completed 

                          ? 'bg-green-50 border-green-200' 

                          : isPast(chore.dueDate)

                          ? 'bg-red-50 border-red-200'

                          : 'bg-gray-50 border-gray-200'

                      }`}

                    >

                      <div className="flex items-start justify-between">

                        <div className="flex-1">

                          <h3 className="font-semibold text-gray-800 text-lg">{chore.name}</h3>

                          <div className="flex items-center gap-4 mt-2 text-sm">

                            <span className="flex items-center gap-1 text-gray-600">

                              <Calendar size={16} />

                              Due: {formatDate(chore.dueDate)}

                            </span>

                            <span className={`px-2 py-1 rounded-full text-xs font-medium ${

                              chore.frequency === 'daily' 

                                ? 'bg-blue-100 text-blue-700' 

                                : 'bg-purple-100 text-purple-700'

                            }`}>

                              {chore.frequency}

                            </span>

                            {chore.recurrence && (

                              <span className="text-gray-500 text-xs">

                                ({chore.recurrence})

                              </span>

                            )}

                          </div>

                          {chore.completed && (

                            <div className="mt-2 text-sm text-green-600 font-medium">

                              ✓ Completed by {chore.completedBy}

                            </div>

                          )}

                        </div>

                        

                        <div className="flex gap-2">

                          {!chore.completed && isToday(chore.dueDate) && (

                            <>

                              <button

                                onClick={() => markAsComplete(chore.id, 'Kawika')}

                                className="bg-teal-500 text-white py-2 px-4 rounded-xl font-medium hover:bg-teal-600 transition text-sm"

                              >

                                Kawika

                              </button>

                              <button

                                onClick={() => markAsComplete(chore.id, 'Jordyn')}

                                className="bg-white border-2 border-teal-500 text-teal-600 py-2 px-4 rounded-xl font-medium hover:bg-teal-50 transition text-sm"

                              >

                                Jordyn

                              </button>

                            </>

                          )}

                          {activeTab === 'upcoming' && (

                            <button

                              onClick={() => {

                                if (expandedCalendarChoreId === chore.id) {

                                  setExpandedCalendarChoreId(null);

                                  setEditingUpcomingDate('');

                                } else {

                                  setExpandedCalendarChoreId(chore.id);

                                  setEditingUpcomingDate(chore.dueDate);

                                }

                              }}

                              className="bg-orange-500 text-white py-2 px-4 rounded-xl font-medium hover:bg-orange-600 transition text-sm flex items-center gap-1"

                            >

                              <Calendar size={16} />

                              Reschedule

                            </button>

                          )}

                          <button

                            onClick={() => deleteChore(chore.id)}

                            className="bg-red-500 text-white p-2 rounded-xl hover:bg-red-600 transition"

                          >

                            <Trash2 size={20} />

                          </button>

                        </div>

                      </div>


                      {/* Reschedule date picker */}

                      {expandedCalendarChoreId === chore.id && activeTab === 'upcoming' && (

                        <div className="mt-4 pt-4 border-t border-gray-200">

                          <label className="block text-sm font-medium text-gray-700 mb-2">

                            Select new due date:

                          </label>

                          <div className="flex gap-2">

                            <input

                              type="date"

                              value={editingUpcomingDate}

                              onChange={(e) => setEditingUpcomingDate(e.target.value)}

                              className="flex-1 px-3 py-2 border border-gray-300 rounded-xl focus:ring-2 focus:ring-teal-500 focus:border-transparent"

                            />

                            <button

                              onClick={() => updateDueDate(chore.id, editingUpcomingDate)}

                              className="bg-orange-500 text-white py-2 px-4 rounded-xl font-medium hover:bg-orange-600 transition"

                            >

                              Update

                            </button>

                          </div>

                        </div>

                      )}

                    </div>

                  ))}

                </div>

              )}

            </div>


            {/* Change Log */}

            <div className="bg-white rounded-2xl shadow-md p-6">

              <h2 className="text-2xl font-bold text-gray-800 mb-4">Recent Activity</h2>

              <div className="space-y-2">

                {changeLog.slice(0, 10).map(log => (

                  <div key={log.id} className="p-3 bg-gray-50 rounded-lg">

                    <div className="flex justify-between items-start">

                      <div>

                        <span className={`inline-block px-2 py-1 rounded text-xs font-medium mr-2 ${

                          log.action === 'completed' ? 'bg-green-100 text-green-700' :

                          log.action === 'added' ? 'bg-blue-100 text-blue-700' :

                          log.action === 'deleted' ? 'bg-red-100 text-red-700' :

                          'bg-orange-100 text-orange-700'

                        }`}>

                          {log.action}

                        </span>

                        <span className="text-gray-800 font-medium">{log.choreName}</span>

                        {log.person !== 'System' && (

                          <span className="text-gray-600 text-sm"> by {log.person}</span>

                        )}

                      </div>

                      <span className="text-gray-500 text-xs">

                        {new Date(log.timestamp).toLocaleString()}

                      </span>

                    </div>

                    {log.details && (

                      <div className="text-sm text-gray-600 mt-1">{log.details}</div>

                    )}

                  </div>

                ))}

              </div>

            </div>

          </div>


          {/* Add Chore Modal */}

          {showAddModal && (

            <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center p-4 z-50">

              <div className="bg-white rounded-2xl p-6 max-w-md w-full">

                <div className="flex justify-between items-center mb-4">

                  <h3 className="text-2xl font-bold text-gray-800">Add New Chore</h3>

                  <button

                    onClick={() => setShowAddModal(false)}

                    className="text-gray-500 hover:text-gray-700"

                  >

                    <X size={24} />

                  </button>

                </div>


                <div className="space-y-4">

                  <div>

                    <label className="block text-sm font-medium text-gray-700 mb-2">

                      Chore Name

                    </label>

                    <input

                      type="text"

                      value={newChore.name}

                      onChange={(e) => setNewChore({ ...newChore, name: e.target.value })}

                      className="w-full px-4 py-2 border border-gray-300 rounded-xl focus:ring-2 focus:ring-teal-500 focus:border-transparent"

                      placeholder="e.g., Wash dishes"

                    />

                  </div>


                  <div>

                    <label className="block text-sm font-medium text-gray-700 mb-2">

                      Frequency

                    </label>

                    <select

                      value={newChore.frequency}

                      onChange={(e) => setNewChore({ ...newChore, frequency: e.target.value })}

                      className="w-full px-4 py-2 border border-gray-300 rounded-xl focus:ring-2 focus:ring-teal-500 focus:border-transparent"

                    >

                      <option value="daily">Daily</option>

                      <option value="weekly">Weekly</option>

                    </select>

                  </div>


                  {newChore.frequency === 'daily' && (

                    <div>

                      <label className="block text-sm font-medium text-gray-700 mb-2">

                        Recurrence

                      </label>

                      <select

                        value={newChore.recurrence}

                        onChange={(e) => setNewChore({ ...newChore, recurrence: e.target.value })}

                        className="w-full px-4 py-2 border border-gray-300 rounded-xl focus:ring-2 focus:ring-teal-500 focus:border-transparent"

                      >

                        <option value="everyday">Every day</option>

                        <option value="weekdays">Weekdays only</option>

                      </select>

                    </div>

                  )}


                  <div>

                    <label className="block text-sm font-medium text-gray-700 mb-2">

                      Due Date

                    </label>

                    <input

                      type="date"

                      value={newChore.dueDate}

                      onChange={(e) => setNewChore({ ...newChore, dueDate: e.target.value })}

                      className="w-full px-4 py-2 border border-gray-300 rounded-xl focus:ring-2 focus:ring-teal-500 focus:border-transparent"

                    />

                  </div>


                  <div className="flex gap-3 pt-4">

                    <button

                      onClick={() => setShowAddModal(false)}

                      className="flex-1 bg-gray-200 text-gray-700 py-3 rounded-xl font-medium hover:bg-gray-300 transition"

                    >

                      Cancel

                    </button>

                    <button

                      onClick={addChore}

                      className="flex-1 bg-teal-500 text-white py-3 rounded-xl font-medium hover:bg-teal-600 transition"

                    >

                      Add Chore

                    </button>

                  </div>

                </div>

              </div>

            </div>

          )}

        </div>

      );

    };


    ReactDOM.render(<ChoreTracker />, document.getElementById('root'));

  </script>
